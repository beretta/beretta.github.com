(function () {
  function activate(id) {
    document.querySelectorAll('.pub-tab').forEach(function (t) {
      var on = t.dataset.tab === id;
      t.classList.toggle('active', on);
      t.setAttribute('aria-selected', on);
    });
    document.querySelectorAll('.pub-pane').forEach(function (p) {
      p.classList.toggle('active', p.id === id);
    });
  }
  var hash = window.location.hash.slice(1);
  var valid = ['journals', 'conferences', 'abstracts', 'books', 'editorials'];
  if (hash && valid.indexOf(hash) !== -1) activate(hash);
  document.querySelectorAll('.pub-tab').forEach(function (tab) {
    tab.addEventListener('click', function () {
      activate(this.dataset.tab);
      history.pushState(null, null, '#' + this.dataset.tab);
    });
  });
})();
